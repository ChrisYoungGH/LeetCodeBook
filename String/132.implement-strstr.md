## 找到子串的起始位置
### 解法
每个位置匹配
```c++
class Solution {
public:
    int strStr(string haystack, string needle) {
        int m = haystack.size(), n = needle.size();
        if (!n) {
            return 0;
        }
        if (!m || n > m) {
            return -1;
        }
        
        for (int i = 0; i <= m - n; i++) {
            bool flag = true;
            for (int j = 0; j < n; j++) {
                if (haystack[i+j] != needle[j]) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                return i;
            }
        }
        
        return -1;
    }
};
```
