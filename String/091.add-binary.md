## 字符串表示的两个二进制数相加
### 解法
```c++
class Solution {
public:
    string addBinary(string a, string b) {
        string res;
        
        int na = a.length(), nb = b.length();
        if (!na || !nb) {
            return res;
        }
        
        int n = max(na, nb);
        int carry = 0;
        
        res = na > nb ? a : b;
        int m = n-1;
        
        for (int i = na-1, j = nb - 1; i >= 0 || j >= 0; i--, j--) {
            int ia = i >= 0 ? a[i] - '0' : 0;
            int ib = j >= 0 ? b[j] - '0' : 0;
            int sum = ia + ib + carry;
            res[m--] = sum % 2 + '0';
            carry = sum / 2;
        }
        
        if (carry) res = '1' + res;
        
        return res;
    }
};
```
