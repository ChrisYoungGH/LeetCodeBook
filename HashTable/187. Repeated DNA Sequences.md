## 出现多次的子序列
### 解法: hash
```c++
class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        unordered_map<string, int> seqCnt;
        vector<string> res;
        int n = s.length();
        if (n < 10) {
            return res;
        }
        
        for (int i = 0; i <= n-10; i++) {
            string dna = s.substr(i, 10);
            seqCnt[dna]++;
        }
        
        for (auto it : seqCnt) {
            if (it.second > 1) {
                res.push_back(it.first);
            }
        }
        
        return res;
    }
};
```
