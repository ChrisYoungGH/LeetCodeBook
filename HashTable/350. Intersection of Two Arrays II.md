## 两个数组公共的数(算重复)
### 解法: hash计数
```c++
class Solution {
public:
    vector<int> intersect(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<int, int> count;
        for (int num : nums1) {
            count[num]++;
        }
        
        vector<int> res;
        for (int num : nums2) {
            if (count[num]) {
                res.push_back(num);
                count[num]--;
            }
        }
        
        return res;
    }
};
```
