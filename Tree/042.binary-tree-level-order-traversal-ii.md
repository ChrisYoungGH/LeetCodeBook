## 层序遍历, 下层在前
### 解法
每层插入在首位
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) {
        vector<vector<int> > res;
        if (!root) {
            return res;
        }
        
        vector<int> level;
        queue<TreeNode*> q;
        q.push(root);
        int currLen = 1, nextLen = 0;
        
        while (!q.empty()) {
            TreeNode *node = q.front();
            q.pop();
            level.push_back(node->val);
            currLen--;
            
            if (node->left) {
                q.push(node->left);
                nextLen++;
            }
            if (node->right) {
                q.push(node->right);
                nextLen++;
            }
            
            if (!currLen) {
                res.insert(res.begin(), level);
                level.clear();
                currLen = nextLen;
                nextLen = 0;
            }
        }
        
        return res;
    }
};
```
