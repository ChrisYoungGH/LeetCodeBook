## 判断二叉树是否对称
### 两个栈, 分别左遍历和右遍历
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        if (!root) {
            return true;
        }
        
        queue<TreeNode*> q1, q2;
        q1.push(root);
        q2.push(root);
        
        while (!q1.empty() && !q2.empty()) {
            TreeNode *node1 = q1.front();
            q1.pop();
            TreeNode *node2 = q2.front();
            q2.pop();
            
            if (!node1 && !node2) {
                continue;
            }
            if (!node1 || !node2) {
                return false;
            }
            if (node1->val != node2->val) {
                return false;
            }
            
            q1.push(node1->left);
            q1.push(node1->right);
            
            q2.push(node2->right);
            q2.push(node2->left);
        }
        
        return true;
    }
};
```
