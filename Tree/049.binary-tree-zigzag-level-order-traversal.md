## ZigZag
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int> > res;
        if (!root) {
            return res;
        }
        
        queue<TreeNode*> q;
        q.push(root);
        int currLen = 1, nextLen = 0;
        bool flag = true;
        
        vector<int> level;
        
        while (!q.empty()) {
            TreeNode *node = q.front();
            q.pop();
            currLen--;
            if (flag) {
                level.push_back(node->val);
            }
            else {
                level.insert(level.begin(), node->val);
            }
            
            if (node->left) {
                q.push(node->left);
                nextLen++;
            }
            if (node->right) {
                q.push(node->right);
                nextLen++;
            }
            
            if (!currLen) {
                res.push_back(level);
                level.clear();
                currLen = nextLen;
                nextLen = 0;
                flag = !flag;
            }
        }
        
        return res;
    }
};
```
