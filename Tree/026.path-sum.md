## 树路径
### 解法
深度优先遍历, 遇到符合的路径则返回(或)
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int sum) {
        if (!root) {
            return false;
        }
        sum -= root->val;
        
        if (!root->left && !root->right) {
            if (!sum) {
                return true;
            }
        }
        
        if (hasPathSum(root->left, sum) || hasPathSum(root->right, sum)) {
            return true;
        }
        
        return false;
    }
};
```
