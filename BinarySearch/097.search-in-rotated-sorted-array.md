## 旋转数组查找
### 判断左右两段状态以及target与mid的关系, 共四种情况:
- 左边升序, 落在左边
- 左边升序, 落在右边
- 右边升序, 落在左边
- 右边升序, 落在右边
判断是否落在升序一边, 不是则落在另一边.
```c++
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int n = nums.size();
        if (!n) {
            return -1;
        }
        
        int low = 0, high = n - 1;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (target == nums[mid]) {
                return mid;
            }
            if (target == nums[low]) {
                return low;
            }
            if (target == nums[high]) {
                return high;
            }
            
            if (nums[mid] > nums[low]) {
                if (target < nums[mid] && target > nums[low]) {
                    high = mid - 1;
                }
                else {
                    low = mid + 1;
                }
            }
            
            else {
                if (target > nums[mid] && target < nums[high]) {
                    low = mid + 1;
                }
                else {
                    high = mid - 1;
                }
            }
        }
        
        return nums[low] == target ? low : -1;
    }
};
```
