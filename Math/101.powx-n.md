## Power
### 解法
转化为unsigned int的指数
- 注意底数为0指数非正无定义
- 注意判断浮点数为0的方法
- 注意int转unsigned int是否会溢出
- 二分计算
```c++
class Solution {
public:
    double myPow(double x, int n) {
        if (-1e-7 < x && x < 1e-7 && n <= 0) {
            return 0;
        }
        
        if (n >= 0) {
            return myPowUnsigned(x, n);
        }
        else {
            return 1.0 / myPowUnsigned(x, (unsigned int)(-long(n)));
        }   
    }
    
    double myPowUnsigned(double x, unsigned int n) {
        if (n == 0) {
            return 1;
        }
        
        if (n == 1) {
            return x;
        }
        
        double remain = n % 2 ? x : 1;
        double half = myPowUnsigned(x, n / 2);
        
        return remain * half * half;
    }
};
```
