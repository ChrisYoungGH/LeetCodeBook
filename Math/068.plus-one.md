## 数组加1
### 解法
低位往高位一直加, 处理进位
```c++
class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int n = digits.size();
        int carry = 1;
        for (int i = n - 1; i >= 0; i--) {
            int s = digits[i] + carry;
            digits[i] = s % 10;
            carry = s / 10;
        }
        
        if (carry) {
            digits.insert(digits.begin(), carry);
        }
        
        return digits;
    }
};
```
