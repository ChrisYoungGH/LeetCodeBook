## 删除倒数第k个节点
### 解法
依然是找到倒数第k个节点的前一个节点, 然后更新next指针. 注意删除的是头结点的情况, 此时k == n, curr2 == NULL, 直接返回头结点下一个节点.
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        if (!head || n <= 0) {
            return NULL;
        }
        
        ListNode *curr1 = head, *curr2 = head;
        for (int i = 0; i < n; i++) {
            curr2 = curr2->next;
        }
        if (!curr2) {
            return head->next;
        }
        
        while (curr2->next) {
            curr1 = curr1->next;
            curr2 = curr2->next;
        }
        
        curr1->next = curr1->next ? curr1->next->next : NULL;
        
        return head;
    }
};
```
