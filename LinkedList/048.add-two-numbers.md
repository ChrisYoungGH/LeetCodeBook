## 链表两数相加
### 解法
维护进位, 注意边界情况. 最后进位为1要新开辟一个节点.
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        if (!l1 && !l2) {
            return NULL;
        }
        
        if (!l1) {
            l1 = l2;
            l2 = NULL;
        }
        ListNode *head = l1;
        
        int carry = 0;
        while (l1 && l2) {
            int val = l1->val + l2->val + carry;
            l1->val = val % 10;
            carry = val / 10;
            
            if (!l1->next || !l2->next) {
                break;
            }
            
            l1 = l1->next;
            l2 = l2->next;
        }
        
        l1->next = (l2 && l2->next) ? l2->next : l1->next;
        
        while (l1 && l1->next) {
            l1 = l1->next;
            
            int val = l1->val + carry;
            l1->val = val % 10;
            carry = val / 10;
        }

        if (carry) {
            l1->next = new ListNode(carry);
        }
        
        return head;        
    }
};
```
