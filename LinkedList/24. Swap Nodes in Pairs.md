互换成对节点. 需要三个指针, 两个指向这两个节点, 一个之前前一个节点, 改变三个节点的next指针. 通过添加头部虚节点简化边界检查.
```c++
ListNode *swapPairs(ListNode *head) {
	ListNode *dummy = new ListNode(-1);
	dummy->next = head;

	ListNode *prev= dummy;
	ListNode *first = prev->next;
	if (!first || !first->next) {
		return head;
	}
	ListNode *second = first->next;


	while (prev && first && second) {
		prev->next = second;
		
		ListNode *temp = first;
		first->next = second->next;
		second->next = temp;

		prev = first;
		first = prev->next;
		if (!first || !first->next) {
			break;
		}
		second = first->next;
	}


	head = dummy->next;
	delete [] dummy;

	return head;
}
```
